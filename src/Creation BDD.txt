create table categorie(
    idCat int Primary key,
    nom varchar(255),
    descr varchar(255)
);

create table Annonce(
	idAn int Primary Key,
	titre varchar(255),
    date Date,
    texte varchar(255)
);

create table photo(
    idPhoto int primary key,
	idAn int,
    file varchar(255),
    date Date,
    tailleOctets int,
    FOREIGN KEY (idAn) REFERENCES Annonce(idAn)
);
